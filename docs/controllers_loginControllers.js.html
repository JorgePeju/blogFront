/**
 * 
 * @param {Object} req 
 * @param {Object} res 
 * @returns {object}
 * @throws {error}
 */
const signIn = async (req, res) => {

    const { email, password } = req.body

    try {

        const userCredentials = await signInWithEmailAndPassword(authFb, email, password);
        
        //* res.cookie()
        res.redirect('/admin')

    } catch (error) {

        res.render('../views/signInForm.ejs', {
            error
        });

    }
};
